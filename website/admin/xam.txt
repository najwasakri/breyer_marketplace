    <script>
        const modal = document.getElementById('courseModal');
        const closeBtn = document.getElementsByClassName('close')[0];

        function showAddCourseModal() {
            document.getElementById('modalTitle').textContent = 'Tambah Kursus Baru';
            document.getElementById('courseForm').reset();
            modal.style.display = 'flex';
        }

        function editCourse(id) {
            document.getElementById('modalTitle').textContent = 'Kemaskini Kursus';
            document.getElementById('course_id').value = id;
            
            // Fetch course data using AJAX
            fetch(`get_course.php?id=${id}`)
                .then(response => response.json())
                .then(data => {
                    document.getElementById('course_name').value = data.course_name;
                    document.getElementById('description').value = data.description;
                    document.getElementById('price').value = data.price;
                    document.getElementById('duration').value = data.duration;
                    modal.style.display = 'flex';
                });
        }

        function deleteCourse(id) {
            if (confirm('Adakah anda pasti untuk memadam kursus ini?')) {
                fetch('delete_course.php', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded',
                    },
                    body: `id=${id}`
                })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        location.reload();
                    } else {
                        alert('Error deleting course');
                    }
                });
            }
        }

        closeBtn.onclick = function() {
            modal.style.display = 'none';
        }

        window.onclick = function(event) {
            if (event.target == modal) {
                modal.style.display = 'none';
            }
        }
    </script>
